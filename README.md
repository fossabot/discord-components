## Install

```bash
npm install --save discord-components
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'discord-components'
import 'discord-components/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
